```C++
//Definition for a binary tree node.
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(NULL), right(NULL) {}
};

int maxDepth(TreeNode* root)
{//递归遍历左右子树的深度，返回最大深度
    if(root==NULL){
        return 0;
    }
    int dl = 1, dr = 1;
    if(root->left!=NULL){
        dl+=maxDepth(root->left);
    }
    if(root->right!=NULL){
        dr+=maxDepth(root->right);
    }
    return dl>dr?dl:dr;
}
```
