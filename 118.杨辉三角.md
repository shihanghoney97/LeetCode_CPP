vector<vector<int>> generate(int numRows)
{
    vector<vector<int>> res;
    for(int i=0; i<numRows; i++){
        vector<int> t;
        int j = 0;
        while(j<i+1){
            if(j==0){
                t.push_back(1);
            }else if(j==i){
                t.push_back(1);
            }else{
                t.push_back(res[i][j-1]+res[i][j]);
            }
            j++;
        }
        res.push_back(t);
    }
    return res;
}
