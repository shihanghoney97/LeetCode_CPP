//二分查找
int searchInsert(vector<int>& nums, int target)
{
    if(nums.size()==0){
        return 0;
    }
    int left = 0, right = nums.size()-1;
    int mid;
    while(left<=right){
        mid = (left+right)/2;
        if(nums[mid]==target){
            return mid;
        }
        if(nums[mid]<target){
            left = mid+1;
        }
        if(nums[mid]>target){
            right = mid-1;
        }
    }
    return left;
}
