struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(NULL) {} //结构体的构造函数，初始化当前节点值为x，指针值为空
};

ListNode* mergeTwoLists(ListNode* l1, ListNode* l2)
{//不带头结点
    ListNode *l = new ListNode(0);
    ListNode *p = l1, *q = l2, *r = l;
    while(p!=NULL && q!=NULL){
        if(p->val<=q->val){
            r->next = p;
            r = p;
            p = p->next;
        }else{
            r->next = q;
            r = q;
            q = q->next;
        }
    }
    if(p!=NULL){
        r->next = p;
    }
    if(q!=NULL){
        r->next = q;
    }
    return l->next;
}
